# True if segment (x1,y1)-(x2,y2) and segment (x3,y3)-(x4,y4) have a intersection
def intersect(x1, y1, x2, y2, x3, y3, x4, y4):
    tc1 = (x1 - x2) * (y3 - y1) + (y1 - y2) * (x1 - x3)
    tc2 = (x1 - x2) * (y4 - y1) + (y1 - y2) * (x1 - x4)
    td1 = (x3 - x4) * (y1 - y3) + (y3 - y4) * (x3 - x1)
    td2 = (x3 - x4) * (y2 - y3) + (y3 - y4) * (x3 - x2)
    return tc1*tc2<0 and td1*td2<0

# Count the number of cross points
def crosscount(x):
    pos = dict([(people[i], (x[i*2],x[i*2+1])) for i in range(n_people)])
    total = 0
    for i in range(0, n_link):
        for j in range(i+1, n_link):
            x1, y1 = pos[df_link.iat[i,0]]
            x2, y2 = pos[df_link.iat[i,1]]
            x3, y3 = pos[df_link.iat[j,0]]
            x4, y4 = pos[df_link.iat[j,1]]
            if intersect(x1,y1,x2,y2,x3,y3,x4,y4):
                total += 1
    return total

# Sum of squared errors from desired length  
def dist(x):
    pos = dict([(people[i], (x[i*2],x[i*2+1])) for i in range(n_people)])
    total = 0.0
    num = 0
    for i in range(0, n_people):
        p1 = people[i]
        x1, y1 = pos[p1]
        for j in range(i+1, n_people):
            p2 = people[j]
            x2, y2 = pos[p2]
            d = ((x1-x2)**2+(y1-y2)**2)**0.5
            if near_table.at[p1, p2] > 0:
                total += (d-len_near)**2
            else:
                total += (d-len_far)**2
            num += 1
    return total / num

# cost function
def cost_func(x):
    n_cross = crosscount(x)
    d = dist(x)
    return d + n_cross*200.0
