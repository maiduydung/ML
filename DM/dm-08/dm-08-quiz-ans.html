Manifold learning¶

In [11]:
# PCA
Y = PCA(n_components=n_components).fit_transform(X)
plt.figure(figsize=(10,7))
plt.title("PCA")
for i in range(n_class):
    marker = '${}$'.format(i)
    y1 = Y[ digits==i ]
    plt.scatter(y1[:, 0], y1[:, 1], marker=marker, s=70, label=i)
plt.xlabel('PCA1')
plt.ylabel('PCA2')
plt.show()
